% Generador de índice automático para el índice
% ...general, lista de figuras y lista de tablas
  % [arg 1] nombre del índice en español
  % [arg 2] tipo de índice
  % [arg 3] variable que contiene el nombre de GIA
  \newenvironment{module.index}[3]{
    \clearpage      % asegura el comienzo de una nueva página
    \renewcommand{#3}
      {\LARGE\begin{center} #1 \end{center}} % modifica el nombre por defecto
    \begin{singlespace}
    #2
    \addcontentsline{toc}{chapter}{#1}
  }{
    \end{singlespace}
  }
  
  % Espacio vertical antes del título de los índices
  \setlength{\cftbeforetoctitleskip}{-20pt} % índice general (ref#01)
  \setlength{\cftbeforeloftitleskip}{-20pt} % lista de figuras (ref#01)
  \setlength{\cftbeforelottitleskip}{-20pt} % lista de tablas (ref#01)
  % Espacio vertical después del titulo de los índices
  \setlength{\cftaftertoctitleskip}{30pt} % índice general (ref#02)
  \setlength{\cftafterloftitleskip}{30pt} % lista de figuras (ref#02)
  \setlength{\cftafterlottitleskip}{30pt} % lista de tablas (ref#02)
  
  % Longitud de espaciado (default 4.5)
  \renewcommand{\cftdotsep}{4.5}
  
  % Tipo de espaciado (punto y/o linea)
  \renewcommand{\cftdot}{.}
  
  % Inserta la separación en los capítulos
  \renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}

% Renombra Figure a Figura
\renewcommand{\figurename}{Figura}

% Establece en negrita el número de etiqueta de una figura
% \captionsetup{labelfont=bf}

% Numeración continua de las figuras
\counterwithout{figure}{chapter}

% Nueva funcionalidad de leyenda para las figuraas
  % [arg 1 : optional] descripción corta
  % [arg 2]            descripción larga
  % [arg 3]            fuente
\newcommand{\captionsource}[3]{
  \def\shortcaption{#1}
  \def\captiontext{#2}
  \ifdefempty{\shortcaption}{
    \caption[#2]{#2}
  }{
    \caption[#1]{#2}
  }
  \vspace{-10pt}
  % \caption*{\textbf{Fuente:} #3}
  \caption*{Fuente: #3}
}
% Cambia el nombre de Figure a Figura en el índice de figuras
\renewcommand{\cftfigpresnum}{Figura }

% Establece el ancho necesario de tabulación
\settowidth{\cftfignumwidth}{Figura 99\:}
